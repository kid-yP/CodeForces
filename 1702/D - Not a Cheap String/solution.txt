t = int(input())for _ in range(t):    w = input().strip()    p = int(input())     price = [ord(chr) - 96 for chr in w]    tot = sum(price)     if tot <= p:        print(w)        continue     freq = [0] * 26    for k in w:        freq[ord(k) - 97] += 1     for i in range(len(freq) - 1, -1, -1):        while freq[i] and tot > p:            tot -= i + 1            freq[i] -= 1     result = []    for x in w:        i = ord(x) - 97        if freq[i]:            result.append(x)            freq[i] -= 1     print("".join(result))
