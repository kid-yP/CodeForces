import bisect t = int(input()) for _ in range(t):    n = int(input())    a = [0] + list(map(int, input().split()))     valid = []    ans = 0     for j in range(1, n + 1):        if a[j] < j:            pos = bisect.bisect_left(valid, a[j])            ans += pos            valid.append(j)     print(ans)
