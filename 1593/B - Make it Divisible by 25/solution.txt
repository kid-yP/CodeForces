t = int(input()) for _ in range(t):    s = input()    n = len(s)        ans = float('inf')        targets = ["00", "25", "50", "75"]        for target in targets:        count = 0        j = n - 1                while j >= 0 and s[j] != target[1]:            j -= 1            count += 1                if j < 0:            continue                j -= 1                while j >= 0 and s[j] != target[0]:            j -= 1            count += 1                if j < 0:            continue                ans = min(ans, count)        print(ans)
