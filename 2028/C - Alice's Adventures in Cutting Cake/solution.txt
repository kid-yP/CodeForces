t = int(input()) for _ in range(t):    n, m, v = map(int, input().split())    a = list(map(int, input().split()))     p = [0] * (n+1)    for i in range(n):        p[i+1] = p[i] + a[i]     L = [0] * (n+1)    cur = 0    c = 0    for i in range(n):        cur += a[i]        if cur >= v:            c += 1            cur = 0        L[i+1] = c     R = [0] * (n+1)    cur = 0    c = 0    for i in range(n-1,-1,-1):        cur += a[i]        if cur >= v:            c += 1            cur = 0        R[i] = c     ans = -1    r = 0     for l in range(n+1):        while r < n and L[l] + R[r+1] >= m:            r += 1         if L[l] + R[r] >= m:            ans = max(ans, p[r] - p[l])     print(ans)
